

PROCEDURE dot_product(v1,v2 : ARRAY_OF Float, var ps :float)
VAR 
ps : float :=0;
BEGIN
FOR i FROM 0 TO length(v1)   DO 

ps:= ps + v1[i]*v2[i];


END_FOR
END
